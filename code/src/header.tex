% Creates header for each page
\pagestyle{fancy}
\fancyhf{}

\fancyhead[R]{
    \headerR \hskip 
    \linepagesep 
    \vfootline 
    \thepage
}
\newskip\linepagesep \linepagesep 5pt\relax
\def\vfootline{%
    \begingroup
    	\rule[-10pt]{0.75pt}{25pt}
    \endgroup
}
\def\headerR{%
	\begin{minipage}[]{240pt}
		\hfill \programNameLong
    	\par
    	\hfill \textcolor{black}{\schoolLong}
    \end{minipage}
}

\fancyhead[L]{\headerL}
\def\headerL{
    \begin{minipage}[]{240pt}
        \textbf{\thisDoc} \hfill
    	\par
    	\myFullName \hfill
    \end{minipage}
}

\renewcommand\headrulewidth{0pt}
